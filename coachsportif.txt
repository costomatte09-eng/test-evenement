    function renderCalendar() {

      // Remove old day cells (keep day labels = first 7 children)

      const firstDay = new Date(curYear, curMonth, 1);
      // Monday-first: 0=Mon … 6=Sun
      let startDow = firstDay.getDay(); // 0=Sun

      const daysInMonth = new Date(curYear, curMonth + 1, 0).getDate();

      // Empty cells

        const hasSlots = SLOTS_BY_DOW[dow] && SLOTS_BY_DOW[dow].length > 0;

        cell.className = 'cal-day';

          cell.addEventListener('click', () => selectDate(date, cell));
          // Re-apply cursor hover
          cell.addEventListener('mouseenter', () => { cur.style.width='16px'; cur.style.height='16px'; ring.style.width='52px'; ring.style.height='52px'; ring.style.borderColor='var(--primary)'; });
          cell.addEventListener('mouseleave', () => { cur.style.width='10px'; cur.style.height='10px'; ring.style.width='36px'; ring.style.height='36px'; ring.style.borderColor='rgba(198,93,59,0.5)'; });
        }

        calGrid.appendChild(cell);
      }
    }

    function selectDate(date, cell) {
      // Unselect previous
      document.querySelectorAll('.cal-day.selected').forEach(c => c.classList.remove('selected'));
      cell.classList.add('selected');
      selectedDate = date;
      selectedSlot = null;

      const dow = date.getDay();
      const slots = SLOTS_BY_DOW[dow] || [];
      const slotsSection = document.getElementById('slotsSection');
      const slotsGrid = document.getElementById('slotsGrid');
      const slotsLabel = document.getElementById('slotsLabel');

      const dayNames = ['Dimanche','Lundi','Mardi','Mercredi','Jeudi','Vendredi','Samedi'];
      slotsLabel.textContent = dayNames[dow] + ' ' + date.getDate() + ' ' + MONTHS[date.getMonth()];

      slotsGrid.innerHTML = '';
      slots.forEach(slot => {
        const btn = document.createElement('button');
        btn.className = 'slot-btn';
        btn.textContent = slot;
        if (FULL_SLOTS.includes(slot)) { btn.classList.add('full'); btn.textContent = slot + ' — complet'; }
        else {
          btn.addEventListener('click', () => selectSlot(slot, btn));
          btn.addEventListener('mouseenter', () => { cur.style.width='16px'; cur.style.height='16px'; ring.style.width='52px'; ring.style.height='52px'; ring.style.borderColor='var(--primary)'; });
          btn.addEventListener('mouseleave', () => { cur.style.width='10px'; cur.style.height='10px'; ring.style.width='36px'; ring.style.height='36px'; ring.style.borderColor='rgba(198,93,59,0.5)'; });
        }
        slotsGrid.appendChild(btn);
      });

      slotsSection.style.display = 'flex';
      document.getElementById('bookingForm').style.display = 'none';
      document.getElementById('bookingCta').style.display = 'none';

      gsap.fromTo(slotsSection, { opacity:0, y:10 }, { opacity:1, y:0, duration:0.4, ease:'power2.out' });
    }

    function selectSlot(slot, btn) {
      document.querySelectorAll('.slot-btn.selected').forEach(b => b.classList.remove('selected'));
      btn.classList.add('selected');
      selectedSlot = slot;

      const form = document.getElementById('bookingForm');
      const cta = document.getElementById('bookingCta');
      form.style.display = 'flex';
      cta.style.display = 'block';
      gsap.fromTo([form, cta], { opacity:0, y:10 }, { opacity:1, y:0, duration:0.4, stagger:0.1, ease:'power2.out' });
    }

    document.getElementById('confirmBtn').addEventListener('click', () => {
      const name = document.getElementById('bfName').value.trim();
      const email = document.getElementById('bfEmail').value.trim();
      if (!name || !email || !selectedDate || !selectedSlot) {
        gsap.fromTo('#bookingForm', { x:0 }, { x:[0,8,-8,6,-4,0], duration:0.4, ease:'none' });
        return;
      }
      const dayNames = ['Dimanche','Lundi','Mardi','Mercredi','Jeudi','Vendredi','Samedi'];
      const dateStr = dayNames[selectedDate.getDay()] + ' ' + selectedDate.getDate() + ' ' + MONTHS[selectedDate.getMonth()] + ' à ' + selectedSlot;
      document.getElementById('confirmDate').textContent = dateStr;

      // Show confirmation, hide rest
      document.getElementById('calGrid').parentElement.style.opacity = '0.35';
      document.getElementById('slotsSection').style.display = 'none';
      document.getElementById('bookingForm').style.display = 'none';
      document.getElementById('bookingCta').style.display = 'none';
      const conf = document.getElementById('bookingConfirm');
      conf.style.display = 'flex';
      gsap.fromTo(conf, { opacity:0, scale:0.95 }, { opacity:1, scale:1, duration:0.5, ease:'back.out(1.5)' });
    });

    document.getElementById('calPrev').addEventListener('click', () => {
      curMonth--; if (curMonth < 0) { curMonth = 11; curYear--; }
      renderCalendar();
      selectedDate = null; selectedSlot = null;
      document.getElementById('slotsSection').style.display = 'none';
      document.getElementById('bookingForm').style.display = 'none';
      document.getElementById('bookingCta').style.display = 'none';
    });
    document.getElementById('calNext').addEventListener('click', () => {
      curMonth++; if (curMonth > 11) { curMonth = 0; curYear++; }
      renderCalendar();
      selectedDate = null; selectedSlot = null;
      document.getElementById('slotsSection').style.display = 'none';
      document.getElementById('bookingForm').style.display = 'none';
      document.getElementById('bookingCta').style.display = 'none';
    });

    renderCalendar();
